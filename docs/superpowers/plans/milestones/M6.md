## MILESTONE 6 — Driver Docker-rootless plugável

Objetivo: um **segundo backend real** por trás da mesma interface `Driver`, dirigindo um
`dockerd-rootless` **por sessão** com **data-root e socket unix próprios** (nunca o
`/var/run/docker.sock` do host, nunca o daemon rootless compartilhado do usuário em
`$XDG_RUNTIME_DIR/docker.sock`). O contrato exposto ao agente é **idêntico** ao do Podman:
mesmos códigos de erro, mesmas labels (`sbx.session`, `sbx.managed`), mesmos comandos
`create/exec/logs/status/destroy`. A paridade é travada extraindo o comportamento numa
**suíte de contrato compartilhada** (`drivertest.RunContract`) rodada contra fake, podman e
docker. Selecionável via `SBX_DRIVER=docker` (env interno; nunca vaza no contrato).

**Pré-requisito de ambiente (integração):** `docker` (CLI) instalado **e** rootless
configurado — `dockerd-rootless.sh` no PATH (via `dockerd-rootless-setuptool.sh install`),
`newuidmap`/`newgidmap` presentes, faixas `/etc/subuid`+`/etc/subgid` para o usuário; em
WSL2, `systemd=true` em `/etc/wsl.conf`. Task 6.3 valida isso no `Preflight`.

**Estratégia de teste (idêntica a M1):** builders de argv/env são funções puras testadas
deterministicamente (sem daemon). Testes que precisam do daemon ficam atrás de
`//go:build integration` + `t.Skip` se `docker`/`dockerd-rootless.sh` ausentes; rodam com
`go test -tags integration ./...`. O guard de segurança "nunca referencia o socket do host"
é **espelhado** para o driver Docker (Task 6.2).

> **Assunção de ordenação com M2 (explícita):** no mapa de dependências das convenções, M6
> **espelha o contrato de M1** e é independente de M2. Este backlog assume que M6 roda com o
> código no estado de **fim de M1** (single-container; `EnvSpec.ComposePath` **rejeitado**).
> Se M2 já tiver aterrissado quando M6 for executado (compose via `docker compose`,
> `EnvSpec.EnvVars`/`Networks`, portas dinâmicas lidas de volta), o driver Docker DEVE honrar
> o **mesmo `EnvSpec`** — `docker compose -p <namespace> -f <file> up -d`, injeção de env
> vars, redes extras, `-P` + `docker inspect --format '{{json .NetworkSettings.Ports}}'` — em
> paridade exata com o Podman de M2. Enquanto M2 não existir, o driver Docker **rejeita
> compose com o mesmo código `compose_unsupported`** (Task 6.4) e o `EnvSpec` estendido é
> ignorado com segurança (campos zero-value). Não implemente compose aqui: isso pertence à
> reabertura de M2/M6 depois de medir o `docker compose ps --format json` rodando.

**Invariantes de segurança preservados (§2 das convenções):** (1) o socket do Docker do host
NUNCA é referenciado/exposto — o cliente fala só com `-H unix://<stateDir>/docker.sock`;
(2) o daemon roda com `--data-root/--exec-root` **próprios** por sessão, nunca o engine do
host; (3) o backend não vaza (seleção via `SBX_DRIVER`, sem flag `--driver`); (6) portas
sempre dinâmicas (herdado de M2 quando aterrissar; em M1-level não há publicação de porta).

```go
// sbx/internal/driver/drivertest/contract.go
package drivertest

import (
	"context"
	"testing"

	"github.com/gustavocarvalho/sbx/internal/driver"
	"github.com/stretchr/testify/require"
)

// RunContract exercises the behavior EVERY Driver must share, so the
// agent-facing CLI contract is identical across engines (fake/podman/docker).
//
// newDriver returns a ready-to-use Driver. It is called ONCE and reused across
// sub-tests, which stay isolated by using a distinct session id each; this
// avoids spinning a fresh rootless daemon per case on the real engines. Only
// cross-engine invariants are asserted — engine-specific details (exact name
// format, exec stdout, non-zero exit passthrough, compose rejection) live in
// each driver's own tests, because the Fake always returns ExitCode 0 and
// ignores EnvSpec.
func RunContract(t *testing.T, newDriver func() driver.Driver) {
	t.Helper()
	d := newDriver()
	ctx := context.Background()

	t.Run("create_status_list_destroy", func(t *testing.T) {
		const sess = "contract-lifecycle"
		e, err := d.Create(ctx, sess, driver.EnvSpec{})
		require.NoError(t, err)
		require.NotEmpty(t, e.ID)
		require.NotEmpty(t, e.Name)
		require.NotEmpty(t, e.Namespace)
		require.Equal(t, "running", e.Status)

		st, err := d.Status(ctx, e.ID)
		require.NoError(t, err)
		require.Equal(t, "running", st.Status)

		list, err := d.List(ctx, sess)
		require.NoError(t, err)
		require.Len(t, list, 1)
		require.Equal(t, e.ID, list[0].ID)

		require.NoError(t, d.Destroy(ctx, e.ID))
		list, err = d.List(ctx, sess)
		require.NoError(t, err)
		require.Len(t, list, 0)
	})

	t.Run("list_filters_by_session", func(t *testing.T) {
		a, err := d.Create(ctx, "contract-sessA", driver.EnvSpec{})
		require.NoError(t, err)
		defer d.Destroy(ctx, a.ID)
		b, err := d.Create(ctx, "contract-sessB", driver.EnvSpec{})
		require.NoError(t, err)
		defer d.Destroy(ctx, b.ID)

		la, err := d.List(ctx, "contract-sessA")
		require.NoError(t, err)
		require.Len(t, la, 1)
		require.Equal(t, a.ID, la[0].ID)

		lb, err := d.List(ctx, "contract-sessB")
		require.NoError(t, err)
		require.Len(t, lb, 1)
		require.Equal(t, b.ID, lb[0].ID)
	})

	t.Run("sequence_distinct_names_same_session", func(t *testing.T) {
		const sess = "contract-seq"
		e1, err := d.Create(ctx, sess, driver.EnvSpec{})
		require.NoError(t, err)
		defer d.Destroy(ctx, e1.ID)
		e2, err := d.Create(ctx, sess, driver.EnvSpec{})
		require.NoError(t, err)
		defer d.Destroy(ctx, e2.ID)

		require.NotEqual(t, e1.ID, e2.ID)
		require.NotEqual(t, e1.Name, e2.Name)

		list, err := d.List(ctx, sess)
		require.NoError(t, err)
		require.Len(t, list, 2)
	})

	t.Run("exec_success_is_zero_exit", func(t *testing.T) {
		const sess = "contract-exec"
		e, err := d.Create(ctx, sess, driver.EnvSpec{})
		require.NoError(t, err)
		defer d.Destroy(ctx, e.ID)

		r, err := d.Exec(ctx, e.ID, []string{"echo", "hi"})
		require.NoError(t, err)
		require.Equal(t, 0, r.ExitCode)
	})

	t.Run("logs_ok", func(t *testing.T) {
		const sess = "contract-logs"
		e, err := d.Create(ctx, sess, driver.EnvSpec{})
		require.NoError(t, err)
		defer d.Destroy(ctx, e.ID)

		_, err = d.Logs(ctx, e.ID, driver.LogOpts{})
		require.NoError(t, err)
	})

	t.Run("destroy_unknown_errors", func(t *testing.T) {
		require.Error(t, d.Destroy(ctx, "sbx-does-not-exist-999"))
	})
}
```

		_ = exec.Command("podman", "--root", dir+"/storage", "--runroot", dir+"/runroot", "system", "reset", "--force").Run()

	})
	return p
}

// sbx/internal/driver/docker.go
package driver

import (
	"context"
	"os/exec"
	"path/filepath"
)

// Docker drives a PER-SESSION rootless dockerd with its OWN data-root and unix
// socket under stateDir. It NEVER touches the host daemon socket
// (/var/run/docker.sock) NOR the user's shared rootless socket
// ($XDG_RUNTIME_DIR/docker.sock): every client call goes through
// -H unix://<stateDir>/docker.sock, and the daemon is launched with
// --data-root/--exec-root under stateDir. Unlike podman (daemonless), docker
// needs a long-lived daemon, so the driver starts one on demand, reuses it
// across `sbx` invocations for the same session, and tears it down via Stop()
// (Task 6.3).
type Docker struct {
	bin        string // "docker" (client)
	daemonBin  string // "dockerd-rootless.sh" (rootless daemon launcher)
	dataRoot   string // <stateDir>/data  (dockerd --data-root)
	execRoot   string // <stateDir>/exec  (dockerd --exec-root)
	sock       string // <stateDir>/docker.sock (our per-session client/daemon socket)
	pidfile    string // <stateDir>/docker.pid
	runtimeDir string // <stateDir>/run   (XDG_RUNTIME_DIR for the daemon)
	logfile    string // <stateDir>/dockerd.log
}

func NewDocker(stateDir string) *Docker {
	return &Docker{
		bin:        "docker",
		daemonBin:  "dockerd-rootless.sh",
		dataRoot:   filepath.Join(stateDir, "data"),
		execRoot:   filepath.Join(stateDir, "exec"),
		sock:       filepath.Join(stateDir, "docker.sock"),
		pidfile:    filepath.Join(stateDir, "docker.pid"),
		runtimeDir: filepath.Join(stateDir, "run"),
		logfile:    filepath.Join(stateDir, "dockerd.log"),
	}
}

func (d *Docker) Name() string { return "docker" }

// baseArgs points the docker CLIENT at OUR per-session daemon socket, never the
// host's /var/run/docker.sock nor the user's shared rootless socket.
func (d *Docker) baseArgs() []string {
	return []string{"-H", "unix://" + d.sock}
}

// daemonArgs are forwarded to dockerd (via dockerd-rootless.sh) so the daemon
// keeps its WHOLE state under stateDir and listens ONLY on our socket.
func (d *Docker) daemonArgs() []string {
	return []string{
		"--data-root", d.dataRoot,
		"--exec-root", d.execRoot,
		"--host", "unix://" + d.sock,
		"--pidfile", d.pidfile,
	}
}

// Preflight (part 1): binary detection only. Daemon bring-up + rootless
// verification are added in Task 6.3.
func (d *Docker) Preflight(ctx context.Context) error {
	if _, err := exec.LookPath(d.bin); err != nil {
		return CLIErrorLike("engine_missing",
			"docker not found on PATH",
			"install docker and the rootless extras; run `dockerd-rootless-setuptool.sh install` once as your user")
	}
	if _, err := exec.LookPath(d.daemonBin); err != nil {
		return CLIErrorLike("engine_missing",
			"dockerd-rootless.sh not found on PATH",
			"install rootless docker: `dockerd-rootless-setuptool.sh install`; ensure the docker rootless bin dir (e.g. ~/bin) is on PATH")
	}
	return nil
}
```

### Task 6.3: Ciclo de vida do `dockerd-rootless` por sessão + `Preflight` verifica rootless

Docker **não é daemonless**: o driver precisa de um daemon vivo que sobreviva entre invocações
`sbx env create` … `sbx env exec` da mesma sessão. `ensureDaemon` sobe um `dockerd-rootless`
**destacado** (`setsid`), reusa o que já estiver de pé, e o `Stop()` derruba tudo no fim
(chamado pelo supervisor de M4 / `sbx session end`; até M4, o harness de integração chama no
cleanup).

- Modify: `sbx/internal/driver/docker.go` (adiciona `mu`, `daemonExtraEnv`, `output`, `daemonResponds`, `ensureDaemon`, `Stop`; estende `Preflight`)

- Create: `sbx/internal/driver/docker_integration_test.go` (`//go:build integration`)

**Interfaces:**
- Produces: `func (d *Docker) ensureDaemon(ctx) error` (idempotente; sobe/reusa o daemon por sessão); `func (d *Docker) Stop() error` (derruba o daemon da sessão); `daemonExtraEnv() []string` (env extra do daemon: `XDG_RUNTIME_DIR`/`DOCKER_HOST` sob `stateDir`); `output(ctx, args) (string, error)` (stdout limpo, stderr separado, como o `cmd.Output()` do Podman). `Preflight` passa a subir o daemon e exigir rootless.

// append to sbx/internal/driver/docker.go

// daemonExtraEnv isolates the daemon's runtime dir and default DOCKER_HOST
// under stateDir, so nothing leaks to the host's $XDG_RUNTIME_DIR/docker.sock.
func (d *Docker) daemonExtraEnv() []string {
	return []string{
		"XDG_RUNTIME_DIR=" + d.runtimeDir,
		"DOCKER_HOST=unix://" + d.sock,
	}
}

// output runs the docker CLIENT and returns clean stdout (stderr captured
// separately), mirroring the podman driver's cmd.Output() usage so daemon
// warnings never contaminate parsed output.
func (d *Docker) output(ctx context.Context, args []string) (string, error) {
	cmd := exec.CommandContext(ctx, d.bin, args...)
	var errb strings.Builder
	cmd.Stderr = &errb
	out, err := cmd.Output()
	if err != nil {
		return strings.TrimSpace(errb.String()), err
	}
	return string(out), nil
}

// daemonResponds returns true when our per-session daemon answers on our socket.
func (d *Docker) daemonResponds(ctx context.Context) bool {
	c := exec.CommandContext(ctx, d.bin, append(d.baseArgs(), "version", "--format", "{{.Server.Version}}")...)
	return c.Run() == nil
}

// ensureDaemon starts (or reuses) the per-session rootless dockerd. It is
// idempotent and cheap when the daemon is already up (one `docker version`).
func (d *Docker) ensureDaemon(ctx context.Context) error {
	d.mu.Lock()
	defer d.mu.Unlock()
	if d.daemonResponds(ctx) {
		return nil // reuse the already-running per-session daemon
	}
	for _, dir := range []string{d.dataRoot, d.execRoot, d.runtimeDir, filepath.Dir(d.sock)} {
		if err := os.MkdirAll(dir, 0o700); err != nil {
			return CLIErrorLike("engine_broken",
				"cannot create docker state dir: "+err.Error(),
				"check permissions on the session state dir")
		}
	}
	logf, err := os.OpenFile(d.logfile, os.O_CREATE|os.O_WRONLY|os.O_APPEND, 0o600)
	if err != nil {
		return CLIErrorLike("engine_broken",
			"cannot open dockerd log: "+err.Error(),
			"check permissions on the session state dir")
	}
	// NOTE: not exec.CommandContext — the daemon must OUTLIVE this short-lived
	// CLI process and be reused by the next `sbx` invocation for the same
	// session. Cancelling ctx must not kill it.
	cmd := exec.Command(d.daemonBin, d.daemonArgs()...)
	cmd.Env = append(os.Environ(), d.daemonExtraEnv()...)
	cmd.Stdout, cmd.Stderr = logf, logf
	// Detach into its own session/process group so it survives the parent and
	// so Stop() can signal the whole group (rootlesskit + dockerd).
	cmd.SysProcAttr = &syscall.SysProcAttr{Setsid: true}
	if err := cmd.Start(); err != nil {
		return CLIErrorLike("engine_broken",
			"failed to start dockerd-rootless: "+err.Error(),
			"run `dockerd-rootless.sh` manually to see the error; ensure newuidmap/newgidmap and /etc/subuid+/etc/subgid are configured")
	}
	deadline := time.Now().Add(30 * time.Second)
	for time.Now().Before(deadline) {
		if d.daemonResponds(ctx) {
			return nil
		}
		time.Sleep(300 * time.Millisecond)
	}
	return CLIErrorLike("engine_broken",
		"dockerd-rootless did not become ready in time",
		"inspect "+d.logfile+" for the daemon error; verify rootless prerequisites")
}

// Stop tears down the per-session daemon. Called by the M4 supervisor /
// `sbx session end`; until M4, the integration harness calls it in cleanup.
func (d *Docker) Stop() error {
	b, err := os.ReadFile(d.pidfile)
	if err != nil {
		return nil // nothing to stop
	}
	pid, err := strconv.Atoi(strings.TrimSpace(string(b)))
	if err != nil || pid <= 0 {
		return nil
	}
	// dockerd-rootless runs under a setsid-created group; signal the whole
	// group so rootlesskit + dockerd both exit, then the pid directly as backup.
	_ = syscall.Kill(-pid, syscall.SIGTERM)
	_ = syscall.Kill(pid, syscall.SIGTERM)
	_ = os.Remove(d.sock)
	return nil
}

// Preflight (part 2): after binary detection, bring the daemon up and require
// it to be rootless — same posture as the podman driver's not_rootless check.
func (d *Docker) Preflight(ctx context.Context) error {
	if _, err := exec.LookPath(d.bin); err != nil {
		return CLIErrorLike("engine_missing",
			"docker not found on PATH",
			"install docker and the rootless extras; run `dockerd-rootless-setuptool.sh install` once as your user")
	}
	if _, err := exec.LookPath(d.daemonBin); err != nil {
		return CLIErrorLike("engine_missing",
			"dockerd-rootless.sh not found on PATH",
			"install rootless docker: `dockerd-rootless-setuptool.sh install`; ensure the docker rootless bin dir (e.g. ~/bin) is on PATH")
	}
	if err := d.ensureDaemon(ctx); err != nil {
		return err // already a DriverError (engine_broken)
	}
	out, err := d.output(ctx, append(d.baseArgs(), "info", "--format", "{{println .SecurityOptions}}"))
	if err != nil {
		return CLIErrorLike("engine_broken",
			"docker info failed: "+strings.TrimSpace(out),
			"inspect "+d.logfile+"; verify the rootless daemon started")
	}
	if !strings.Contains(out, "rootless") {
		return CLIErrorLike("not_rootless",
			"docker daemon is not running rootless",
			"this tool requires rootless docker for host isolation; do not point it at a root daemon")
	}
	return nil
}

func (d *Docker) Create(ctx context.Context, sessionID string, spec EnvSpec) (Env, error) {
	// Reject compose BEFORE touching the daemon, exactly like the podman driver.
	if spec.ComposePath != "" {
		return Env{}, DriverError{
			Code:    "compose_unsupported",
			Message: "compose (--from) is not supported by the docker driver yet",
			Hint:    "compose support lands in M2 (docker compose); omit --from to create a single-container env",
		}
	}
	if err := d.ensureDaemon(ctx); err != nil {
		return Env{}, err
	}
	image := defaultImage
	if v := spec.Labels["image"]; v != "" {
		image = v
	}
	// Sequence derived from durable state (existing containers for this
	// session), so it survives across CLI invocations — same as podman.
	existing, err := d.List(ctx, sessionID)
	if err != nil {
		return Env{}, err // List already returns a DriverError
	}
	name := naming.EnvName(sessionID, len(existing)+1)
	if _, errs, err := d.run(ctx, d.createArgs(name, sessionID, image)); err != nil {
		return Env{}, DriverError{Code: "create_failed", Message: strings.TrimSpace(errs), Hint: "verify the image name and that rootless docker can pull it"}
	}
	return Env{ID: name, Name: name, Namespace: name, Status: "running"}, nil
}

func (d *Docker) Destroy(ctx context.Context, id string) error {
	if err := d.ensureDaemon(ctx); err != nil {
		return err
	}
	if _, errs, err := d.run(ctx, append(d.baseArgs(), "rm", "-f", id)); err != nil {
		return DriverError{Code: "destroy_failed", Message: strings.TrimSpace(errs), Hint: "id may not exist; run `sbx env status --json`"}
	}
	return nil
}

type dockerPsRow struct {
	Names  string `json:"Names"`
	State  string `json:"State"`
	Labels string `json:"Labels"`
}

// parseDockerPS parses `docker ps --format json` output: NDJSON (one object per
// line), with Names as a comma-joined string. Kept pure so it is unit-tested
// without a daemon.
func parseDockerPS(out string) ([]Env, error) {
	var envs []Env
	for _, line := range strings.Split(strings.TrimSpace(out), "\n") {
		line = strings.TrimSpace(line)
		if line == "" {
			continue
		}
		var r dockerPsRow
		if err := json.Unmarshal([]byte(line), &r); err != nil {
			return nil, DriverError{Code: "parse_failed", Message: err.Error()}
		}
		name := r.Names
		if i := strings.IndexByte(name, ','); i >= 0 {
			name = name[:i] // first name only
		}
		envs = append(envs, Env{ID: name, Name: name, Namespace: name, Status: r.State})
	}
	return envs, nil
}

func (d *Docker) List(ctx context.Context, sessionID string) ([]Env, error) {
	if err := d.ensureDaemon(ctx); err != nil {
		return nil, err
	}
	out, errs, err := d.run(ctx, append(d.baseArgs(), "ps", "-a", "--filter", "label=sbx.session="+sessionID, "--format", "json"))
	if err != nil {
		return nil, DriverError{Code: "list_failed", Message: strings.TrimSpace(errs)}
	}
	return parseDockerPS(out)
}

func (d *Docker) Status(ctx context.Context, id string) (Env, error) {
	if err := d.ensureDaemon(ctx); err != nil {
		return Env{}, err
	}
	out, errs, err := d.run(ctx, append(d.baseArgs(), "inspect", id, "--format", "{{.State.Status}}"))
	if err != nil {
		return Env{}, DriverError{Code: "not_found", Message: strings.TrimSpace(errs), Hint: "run `sbx env status` to list ids"}
	}
	return Env{ID: id, Name: id, Namespace: id, Status: strings.TrimSpace(out)}, nil
}

// append to sbx/internal/driver/docker.go
import (
	"errors"
	"fmt"
)

// Compile-time proof that *Docker satisfies Driver (all methods now exist).
var _ Driver = (*Docker)(nil)

func (d *Docker) Exec(ctx context.Context, id string, cmd []string) (ExecResult, error) {
	if err := d.ensureDaemon(ctx); err != nil {
		return ExecResult{}, err
	}
	args := append(d.baseArgs(), "exec", id)
	args = append(args, cmd...)
	c := exec.CommandContext(ctx, d.bin, args...)
	var out, errb strings.Builder
	c.Stdout, c.Stderr = &out, &errb
	err := c.Run()
	res := ExecResult{Stdout: out.String(), Stderr: errb.String()}
	var ee *exec.ExitError
	if errors.As(err, &ee) {
		res.ExitCode = ee.ExitCode()
		return res, nil // non-zero exit is a valid result, not a driver error
	}
	if err != nil {
		return ExecResult{}, DriverError{Code: "exec_failed", Message: strings.TrimSpace(errb.String()), Hint: "id may not exist or container not running"}
	}
	return res, nil
}

func (d *Docker) logsArgs(id string, opts LogOpts) []string {
	args := append(d.baseArgs(), "logs")
	if opts.Tail > 0 {
		args = append(args, "--tail", fmt.Sprintf("%d", opts.Tail))
	}
	args = append(args, id)
	return args
}

func (d *Docker) Logs(ctx context.Context, id string, opts LogOpts) (string, error) {
	if err := d.ensureDaemon(ctx); err != nil {

// sbx/internal/driver/registry.go
package driver

import "fmt"

// Select builds the Driver named by name (typically sourced from the internal
// SBX_DRIVER env var — never a CLI flag, so the agent-facing contract never
// leaks the backend). stateDir scopes the driver's own storage/socket so it
// never collides with the host's default engine.
func Select(name, stateDir string) (Driver, error) {
	switch name {
	case "", "podman":
		return NewPodman(stateDir), nil
	case "docker":
		return NewDocker(stateDir), nil
	case "fake":
		return NewFake(), nil
	default:
		return nil, DriverError{
			Code:    "unknown_driver",
			Message: fmt.Sprintf("unknown driver %q", name),
			Hint:    "set SBX_DRIVER to podman, docker, or fake",
		}
	}
}
```
